# Examples

## 1. テンプレート利用ランキング集計（関数）

```ts
import { describe, it } from "vitest";

describe("テンプレート利用ランキング集計", () => {
  describe("正常系", () => {
    // 集計対象データが存在する場合にランキングを生成できることを確認する
    it.todo("データが存在する時、利用回数の多い順でランキングが生成されること");

    // 指定期間の集計結果がダッシュボード表示用の形式で返ることを確認する
    it.todo("集計期間を指定した時、/dashboard 表示に必要な項目を含むこと");
  });

  describe("仕様制約", () => {
    // 上位20件の制約を満たすことを確認する
    it.todo("21件以上の結果がある時、ランキング上位20件のみ返すこと");

    // デフォルト期間が過去7日であることを確認する
    it.todo("期間指定がない時、今日から過去7日間を集計範囲にすること");

    // フロント指定期間が優先されることを確認する
    it.todo("フロントで期間指定した時、指定範囲で再集計されること");
  });

  describe("境界値", () => {
    // データが0件でも破綻しないことを確認する
    it.todo("集計範囲にデータが0件の時、空ランキングを返すこと");

    // ちょうど20件の場合の扱いを確認する
    it.todo("結果がちょうど20件の時、20件すべて返すこと");

    // 期間境界日の取り扱いを確認する
    it.todo("開始日と終了日の境界にある利用記録を含めて集計できること");
  });

  describe("並び順・同率", () => {
    // 同率発生時の順位決定ルールを確認する
    it.todo("利用回数が同率の時、仕様で定義された順序で表示されること");

    // 並び順が降順で固定されることを確認する
    it.todo("ランキング表示の時、利用回数の降順で並ぶこと");
  });

  describe("入力バリデーション", () => {
    // 期間の逆転入力に対する扱いを確認する
    it.todo("開始日が終了日より後の時、仕様どおりにエラーまたは補正されること");

    // 不正な期間フォーマットの扱いを確認する
    it.todo("期間フォーマットが不正な時、集計処理を実行しないこと");
  });
});
```

## 2. テンプレート利用ランキングHook（React Hook）

```ts
import { describe, it } from "vitest";

describe("useTemplateUsageRanking", () => {
  describe("初期状態", () => {
    // 初期レンダー時の既定期間を確認する
    it.todo("初回レンダーの時、デフォルト期間が過去7日間であること");

    // 初期表示用データ構造の整合を確認する
    it.todo("初回レンダーの時、ランキングデータが未取得状態で初期化されること");
  });

  describe("状態遷移", () => {
    // 期間変更時の再取得トリガーを確認する
    it.todo("期間を変更した時、最新条件でランキングを再取得すること");

    // 取得成功時の状態反映を確認する
    it.todo("集計取得に成功した時、上位20件までのランキング状態に更新されること");
  });

  describe("異常系", () => {
    // 取得失敗時のエラー状態遷移を確認する
    it.todo("集計取得に失敗した時、エラー状態へ遷移すること");

    // 失敗後の再実行で復帰できることを確認する
    it.todo("再取得に成功した時、エラー状態から復帰すること");
  });

  describe("副作用", () => {
    // アンマウント時の後始末を確認する
    it.todo("アンマウント時、不要な更新が発生しないこと");

    // Strict Mode想定の再実行耐性を確認する
    it.todo("Strict Modeで再実行された時、状態の不整合が発生しないこと");
  });
});
```
